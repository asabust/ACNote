# [847. 图中点的层次](https://www.acwing.com/activity/content/problem/content/910/1/)

> tags: `BFS`

- 所有边的长度都是 1，可以用 BFS 来求最短距离

```c++
#include <cstring>
#include <iostream>
#include <queue>

using namespace std;

const int N = 100010;

int h[N], e[N], ne[N], idx;
int d[N];

int n, m;

void add(int a, int b){
    e[idx] = b;
    ne[idx] = h[a];
    h[a] = idx ++;
}

int bfs(){
    queue<int> q;
    q.push(1);

    memset(d, -1, sizeof d);
    d[1] = 0;

    while(q.size()){
        auto t = q.front();
        q.pop();

        for(int i = h[t]; i != -1; i = ne[i]){
            int j = e[i];
            if(d[j] == -1){
                d[j] = d[t] + 1;
                q.push(j);
            }
        }
    }

    return d[n];
}

int main(){
    cin >> n >> m ;

    memset(h, -1, sizeof h);

    for(int i = 0; i < m; i ++){
        int a, b;
        cin >> a >> b;
        add(a, b);
    }

    cout << bfs() << endl;
    return 0;
}
```
