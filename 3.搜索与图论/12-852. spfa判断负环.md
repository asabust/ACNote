# [852. spfa 判断负环](https://www.acwing.com/problem/content/854/)

[打卡](https://www.acwing.com/activity/content/problem/content/921/1/)

dist[b]：1->b 的最短距离
cnt[b]：这条最短路边数

更新的时候维护 cnt，`dist[b] = dist[a] + w[i]; cnt[b] = cnt[t] + 1;`，一旦`cnt > n`就意味着有负环。

```c++
#include <cstring>
#include <iostream>
#include <queue>

using namespace std;

const int N = 2010, M = 10010;

int h[N], w[M], e[M], ne[M], idx;
int dist[N], cnt[N];
bool st[N];
int m, n;

void add(int a, int b, int c){
    e[idx] = b, w[idx] = c, ne[idx] = h[a], h[a] = idx ++;
}

bool spfa(){

    queue<int> q;
    for(int i = 1; i <= n; i ++){
        q.push(i);
        st[i] = true;
    }

    while(q.size()){
        int t = q.front();
        q.pop();
        st[t] = false;


        for(int i = h[t]; i != -1; i = ne[i]){
            int j = e[i];
            if(dist[j] > dist[t] + w[i]){
                dist[j] = dist[t] + w[i];
                cnt[j] = cnt[t] + 1;

                if(cnt[j] >= n ) return true;
                if(!st[j]){
                    q.push(j);
                    st[j] = true;
                }
            }
        }
    }
    return false;
}

int main(){
    cin >> n >> m;

    memset(h, -1, sizeof h);
    while(m -- ){
        int a, b , c;
        cin >> a >> b >> c;
        add(a, b, c);
    }
    if(spfa()) cout << "Yes" << endl;
    else cout << "No" << endl;
    return 0;
}
```
